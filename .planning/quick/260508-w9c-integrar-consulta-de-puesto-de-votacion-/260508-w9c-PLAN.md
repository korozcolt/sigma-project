**app.py — Flask service on port 5757:**

Structure:
- Import: flask (Flask, request, jsonify), threading, uuid, time, playwright.sync_api (sync_playwright)
- In-memory sessions dict: `sessions = {}` (protected with threading.Lock)
- `POST /lookup` endpoint:
  - Reads JSON body `{"cedula": "..."}`
  - Validates cedula present, returns 400 if missing
  - Generates `session_id = str(uuid.uuid4())`
  - Sets `sessions[session_id] = {"status": "pending", "data": None, "error": None}`
  - Spawns a background thread calling `run_lookup(session_id, cedula)`
  - Returns `{"session_id": session_id}` with 200
- `GET /result/<session_id>` endpoint:
  - Returns `{"status": ..., "data": ..., "error": ...}` from sessions dict
  - Returns 404 if session_id not found
- `run_lookup(session_id, cedula)` function (runs in thread):
  - Opens visible (headless=False) Chromium via sync_playwright
  - Navigates to `https://eleccionescolombia.registraduria.gov.co/identificacion`
  - Fills cedula: try `input[name="nuip"]` first, fallback to `input[type="number"]`, fallback to `input#nuip`
  - Clicks button with text "Consultar" (use `page.get_by_role("button", name="Consultar")`)
  - Sets session status to `"waiting_captcha"`
  - Waits up to 120s for a result element to appear (poll every 2s for selector `.resultado, #resultado, [class*="resultado"], table, .puesto` or for URL to change from the input page)
  - Once result visible, parse the HTML:
    - Extract puesto_nombre: look for element containing "PUESTO" or "Puesto" text, get next sibling or parent text
    - Extract mesa_numero: look for "Mesa" or "MESA"
    - Extract zona_codigo: look for "Zona" or "ZONA"
    - Extract direccion: look for "Dirección" or "DIRECCIÓN"
    - Extract municipio: look for "Municipio" or "MUNICIPIO"
    - Extract departamento: look for "Departamento" or "DEPARTAMENTO"
    - Extract puesto_codigo: first 2 chars of puesto_nombre if it starts with digits, otherwise empty string
    - Strategy: use `page.inner_text("body")` and parse line by line for label:value pairs
  - Sets session to `{"status": "done", "data": {puesto_nombre, puesto_codigo, zona_codigo, mesa_numero, departamento, municipio, direccion}}`
  - On any exception: sets `{"status": "error", "error": str(exception)}`
  - Always closes browser in finally block
- Run: `if __name__ == "__main__": app.run(port=5757, debug=False)`

**3. Create app/Services/RegistraduriaService.php:**

```php
<?php

declare(strict_types=1);

namespace App\Services;

use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class RegistraduriaService
{
    protected string $baseUrl;

    public function __construct()
    {
        $this->baseUrl = config('services.registraduria.url', 'http://localhost:5757');
    }

    /**
     * Start an async lookup for the given cedula.
     * Returns the session_id from the Python service.
     *
     * @throws \Exception if the service is unreachable or returns an error
     */
    public function startLookup(string $cedula): string
    {
        $response = Http::timeout(10)
            ->post("{$this->baseUrl}/lookup", ['cedula' => $cedula]);

        if (! $response->successful()) {
            Log::error('RegistraduriaService: lookup failed', [
                'cedula' => $cedula,
                'status' => $response->status(),
                'body' => $response->body(),
            ]);
            throw new \Exception('El servicio de Registraduría no está disponible. Inicia el servicio Python primero.');
        }

        $sessionId = $response->json('session_id');

        if (! $sessionId) {
            throw new \Exception('El servicio de Registraduría no devolvió un session_id válido.');
        }

        return $sessionId;
    }

    /**
     * Poll the result for a given session.
     *
     * Returns array with keys: status, data, error
     * status: pending | waiting_captcha | done | error
     * data: puesto_nombre, puesto_codigo, zona_codigo, mesa_numero, departamento, municipio, direccion
     */
    public function getResult(string $sessionId): array
    {
        $response = Http::timeout(5)
            ->get("{$this->baseUrl}/result/{$sessionId}");

        if ($response->status() === 404) {
            return ['status' => 'error', 'data' => null, 'error' => 'Sesión no encontrada'];
        }

        if (! $response->successful()) {
            return ['status' => 'error', 'data' => null, 'error' => 'Error comunicándose con el servicio'];
        }

        return $response->json();
    }
}
```

**Create `app/Filament/Resources/Voters/Concerns/HasRegistraduriaPolling.php`:**

```php
<?php

namespace App\Filament\Resources\Voters\Concerns;

use App\Models\Department;
use App\Models\Municipality;
use App\Models\PollingPlace;
use App\Services\RegistraduriaService;
use Filament\Notifications\Notification;

trait HasRegistraduriaPolling
{
    public function pollRegistraduria(string $sessionId): ?array
    {
        $service = new RegistraduriaService;
        $result = $service->getResult($sessionId);

        if ($result['status'] === 'done' && isset($result['data'])) {
            $data = $result['data'];

            // Resolve municipality by name (case-insensitive match)
            $municipality = Municipality::query()
                ->whereRaw('LOWER(name) = ?', [strtolower($data['municipio'] ?? '')])
                ->first();

            // Resolve department by name if municipality not found
            $department = null;
            if ($municipality) {
                $department = $municipality->department;
            } else {
                $department = Department::query()
                    ->whereRaw('LOWER(name) = ?', [strtolower($data['departamento'] ?? '')])
                    ->first();
            }

            // Find or create PollingPlace
            $placeCode = $data['puesto_codigo'] ?? substr($data['puesto_nombre'] ?? '', 0, 2);
            $pollingPlace = null;

            if ($municipality) {
                $pollingPlace = PollingPlace::query()
                    ->where('municipality_id', $municipality->id)
                    ->where('zone_code', $data['zona_codigo'] ?? null)
                    ->where('place_code', $placeCode)
                    ->first();

                if (! $pollingPlace) {
                    $pollingPlace = PollingPlace::create([
                        'municipality_id' => $municipality->id,
                        'department_id' => $department?->id,
                        'zone_code' => $data['zona_codigo'] ?? null,
                        'place_code' => $placeCode,
                        'name' => $data['puesto_nombre'] ?? 'Desconocido',
                        'address' => $data['direccion'] ?? null,
                    ]);
                }
            }

            // Set form fields via Livewire
            if ($municipality) {
                $this->data['municipality_id'] = $municipality->id;
            }
            if ($pollingPlace) {
                $this->data['polling_place_id'] = $pollingPlace->id;
            }
            $this->data['polling_table_number'] = ltrim($data['mesa_numero'] ?? '', '0') ?: null;

            Notification::make()
                ->title('Puesto de votación encontrado')
                ->body("Puesto: {$data['puesto_nombre']} — Mesa: {$data['mesa_numero']}")
                ->success()
                ->send();

            return $result;
        }

        if ($result['status'] === 'error') {
            Notification::make()
                ->title('Error al consultar Registraduría')
                ->body($result['error'] ?? 'Error desconocido')
                ->danger()
                ->send();

            return $result;
        }

        // Still pending or waiting_captcha — return null to keep polling
        return null;
    }
}
```
